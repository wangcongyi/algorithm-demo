#### Q

```

实现 pow(x, n) ，即计算 x 的 n 次幂函数（即，xn）。
 
```

#### A

```js

const quickPow = (x, n) => {
  let ans = 1
  let _x = x

  while (n > 0) {
    if (n % 2 === 1) {
      ans *= _x
    }
    _x *= _x
    n = Math.floor(n / 2)
  }
  return ans
}


const myPow = (x, n) => {
  if (n >= 0) {
    return quickPow(x, n)
  }
  return 1 / quickPow(x, -n)
}

```
